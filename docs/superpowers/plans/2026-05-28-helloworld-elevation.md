## Task 1: CSS scoping fix — dual selector for class rules

**Files:**
- Modify: `Sources/Swiflow/CSS/CSSSheet.swift:17-37`
- Test: `Tests/SwiflowTests/CSS/CSSSheetTests.swift`

The current behavior emits `.swiflow-T .x` for `rule(".x")`, a descendant selector. The body root *has* the `.x` class but is also the scope class element, so descendant matching misses it. Fix: emit both compound (`.swiflow-T.x`) and descendant (`.swiflow-T .x`) so root and nested matches both work.

- [ ] **Step 3: Implement dual-selector emission**

```swift
case .rule(let selector, let declarations):
    let effectiveSelector: String
    if !shouldScope(selector) {
        effectiveSelector = selector
    } else if selector.hasPrefix(".") {
        // Class-leading selector: emit both the compound form (matches the
        // scope-class root element when it carries this class) and the
        // descendant form (matches nested elements).
        let compound = ".\(scopeClass)\(selector)"
        let descendant = ".\(scopeClass) \(selector)"
        effectiveSelector = "\(compound), \(descendant)"
    } else {
        effectiveSelector = ".\(scopeClass) \(selector)"
    }
    let decls = declarations.map { "  \($0.name): \($0.value);" }.joined(separator: "\n")
    return "\(effectiveSelector) {\n\(decls)\n}"
```

## Task 7: Add `SwiflowWeb.after(_:do:)` cancellable timer

**Files:**

Tiny `setTimeout`-backed helper with a cancel token. Used by Toast for its 2.5s auto-dismiss; cancelled on early unmount.

- [ ] **Step 1: Create the file**

Create `Sources/SwiflowWeb/Timing.swift`:

```swift
#if canImport(JavaScriptKit)
import JavaScriptKit

/// A cancellable scheduled callback. Returned by `after(_:do:)`.
/// Calling `cancel()` clears the underlying `setTimeout` handle. Calling
/// `cancel()` after the timer has already fired is a no-op.
public final class TimerHandle {
    private var handle: JSValue?
    fileprivate init(_ handle: JSValue) { self.handle = handle }

    public func cancel() {
        guard let h = handle else { return }
        _ = JSObject.global.clearTimeout!(h)
        handle = nil
    }
}

/// Schedule `body` to run after `seconds`. Returns a handle; call `cancel()`
/// to prevent the callback from running. Safe to use from `onAppear` and
/// cancel from `onDisappear`.
@MainActor
public func after(_ seconds: Double, do body: @escaping @MainActor () -> Void) -> TimerHandle {
    let closure = JSClosure { _ in
        body()
        return .undefined
    }
    let ms = JSValue.number(seconds * 1000)
    let handle = JSObject.global.setTimeout!(closure, ms)
    return TimerHandle(handle)
}
#endif
```

import Swiflow
import SwiflowWeb
import JavaScriptKit

/// Toast — top-layer notification using the Popover API.
///
/// - `popover="manual"` keeps the toast on the top layer without
///   light-dismiss (other clicks aren't hijacked).
/// - Auto-dismisses after 2.5s via `after(_:do:)`; the timer is cancelled
///   on `onDisappear` so an early parent unmount doesn't fire onDone.
/// - Existing `exitAnimation` / `exitDuration` still drive the exit
///   animation when the parent toggles `showToast = false`.
@MainActor @Component
final class Toast {
    let message: String
    let onDone: () -> Void
    let root = Ref<JSObject>()
    var dismissTimer: TimerHandle?

    init(message: String, onDone: @escaping () -> Void) {
        self.message = message
        self.onDone = onDone
    }

    static var exitAnimation: String? = "toast-out 0.2s ease forwards"
    static var exitDuration: Double?  = 0.2

    var body: VNode {
        div(.attr("popover", "manual"),
            .attr("role", "status"),
            .attr("aria-live", "polite"),
            .ref(root),
            .on(.click) { self.onDone() }) {
            span("✓", .class("icon"), .attr("aria-hidden", "true"))
            text(message)
        }
    }

    func onAppear() {
        if let el = root.wrappedValue {
            _ = el.showPopover?()
        }
        dismissTimer = after(2.5) { [weak self] in self?.onDone() }
    }

    func onDisappear() {
        dismissTimer?.cancel()
        dismissTimer = nil
    }
}

/// Counter — the HelloWorld showcase root.
///
/// Showcases:
/// - `host { }` + dual-selector class rules for scoped CSS that hits the root.
/// - `view-transition-name` on `.count` + `document.startViewTransition` for a
///   morphing increment.

/// - Popover API + anchor positioning for the About card (declarative; no
///   Swift event handler on the trigger).
/// - `<details>` disclosure with animated open/close via `interpolate-size`.
/// - `color-mix` + `light-dark` system colors — auto-themes from OS.
/// - `@container` query on the card via the `raw(...)` escape hatch.
/// - `@property --accent` registered custom property, animated on increment.
@MainActor @Component
final class Counter {
    @State var count: Int = 0
    @State var greeting: String = "Swiflow"
    @State var celebrate: Bool = false
    @State var showToast: Bool = false
    @State var showSignIn: Bool = false
    let greetingInput = Ref<JSObject>()
    let signInDialog = Ref<JSObject>()

    var body: VNode {
        div(.class("card")) {
            header(.class("header")) {
                h1("Hello, \(greeting)!\(celebrate ? " \u{1F389}" : "")",
                   .class("greeting-heading"))
                button("ⓘ",
                       .class("info-trigger"),
                       .attr("popovertarget", "about-popover"),
                       .attr("aria-label", "About Swiflow"))
            }

            p("Count: \(count)",
              .class("count"),
              .attr("aria-live", "polite"))

            div(.class("actions")) {
                button("Increment", .on(.click) { self.increment() })
                button("Show toast", .class("secondary"), .on(.click) { self.showToast = true })
                button("Sign in…", .class("secondary"), .on(.click) { self.openSignIn() })
            }

            div(.class("greeting-row")) {
                label("Greeting:", .attr("for", "g"))
                input(.id("g"), .value($greeting), .ref(greetingInput))
            }

            label(.class("checkbox-row")) {
                input(.attr("type", "checkbox"), .checked($celebrate))
                text(" Celebrate")
            }

            details(.class("inspector")) {
                summary("What's running here?")
                ul(.class("inspector-list")) {
                    li(".count text — animates via the View Transitions API on increment.")
                    li("Sign in… — opens a native <dialog>.")
                    li("ⓘ — opens an `auto` popover anchored via CSS Anchor Positioning.")
                    li("Show toast — mounts a `manual` popover with a 2.5s auto-dismiss.")
                }
            }

            if showToast {
                embed { Toast(message: "Saved!", onDone: { self.showToast = false }) }
            }

            embed { AboutPopover() }

            dialog(.ref(signInDialog),
                   .class("signin-dialog"),
                   .on(.click) { self.handleDialogClick() }) {
                if showSignIn {
                    embed { SignIn(onClose: { self.closeSignIn() }) }
                }
            }
        }
    }

    func onAppear() {
        if let el = greetingInput.wrappedValue { _ = el.focus?() }
    }

    func increment() {
        let document = JSObject.global.document
        let startVT = document.startViewTransition
        if startVT != .undefined {
            let cb = JSClosure { _ in self.count += 1; return .undefined }
            _ = document.object!.startViewTransition!(cb)
        } else {
            count += 1
        }
    }

    func openSignIn() {
        showSignIn = true
        if let el = signInDialog.wrappedValue { _ = el.showModal?() }
    }

    func closeSignIn() {
        if let el = signInDialog.wrappedValue { _ = el.close?() }
        showSignIn = false
    }

    func handleDialogClick() {
        // Backdrop click closes — the dialog itself receives the click when
        // the user clicks outside the rendered content. (The form content
        // catches its own clicks because we don't bubble through it here.)
        // The current event handler shape doesn't expose `event.target`, so
        // we approximate: any direct click on the dialog (not its content)
        // closes via the same path as Cancel/Escape. SignIn's content uses
        // .signin which sits inside the dialog; the dialog is the click target
        // only when the user hits the area outside it.
        closeSignIn()
    }
}

@main
struct App {
    @MainActor
    static func main() {
        Swiflow.render(into: "#app") { Counter() }
    }
}

> **Note on `handleDialogClick`:** the spec lists the backdrop-click handling as an open question. The simplest workable approximation uses `pointer-events: none` on `.signin-dialog` and `pointer-events: auto` on the inner content + `::backdrop` — clicks reach the dialog only via the backdrop. We apply that pattern in the styles file (`pointer-events` on `.signin-dialog`, restored on `.signin`). If a future Swiflow event API exposes `event.target`, this can be tightened.
